module tango.store.impl.CLCell;

private import tango.store.impl.Cell;


/**
 *
 *
 * CLCells are cells that are always arranged in circular lists
 * They are pure implementation tools
 * 
        author: Doug Lea
 * @version 0.93
 *
 * <P> For an introduction to this package see <A HREF="index.html"> Overview </A>.
**/

public class CLCellT(T) : CellT!(T)
{
        // instance variables

        private CLCellT next_;
        private CLCellT prev_;

        // constructors

        /**
         * Make a cell with contents v, previous cell p, next cell n
        **/

        public this (T v, CLCellT p, CLCellT n)
        {
                super(v);
                prev_ = p;
                next_ = n;
        }

        /**
         * Make a singular cell
        **/

        public this (T v)
        {
                super(v);
                prev_ = this;
                next_ = this;
        }

        /**
         * Make a singular cell with null contents
        **/

        public this ()
        {
                super(T.init);
                prev_ = this;
                next_ = this;
        }

        /**
         * return next cell
        **/

        public final CLCellT next()
        {
                return next_;
        }

        /**
         * Set next cell. You probably don't want to call this
        **/

        public final void next(CLCellT n)
        {
                next_ = n;
        }


        /**
         * return previous cell
        **/
        public final CLCellT prev()
        {
                return prev_;
        }

        /**
         * Set previous cell. You probably don't want to call this
        **/
        public final void prev(CLCellT n)
        {
                prev_ = n;
        }


        /**
         * Return true if current cell is the only one on the list
        **/

        public final bool isSingleton()
        {
                return next_ is this;
        }

        public final void linkNext(CLCellT p)
        {
                if (p !is null)
                   {
                   next_.prev_ = p;
                   p.next_ = next_;
                   p.prev_ = this;
                   next_ = p;
                   }
        }

        /**
         * Make a cell holding v and link it immediately after current cell
        **/

        public final void addNext(T v)
        {
                CLCellT p = new CLCellT(v, this, next_);
                next_.prev_ = p;
                next_ = p;
        }

        /**
         * make a node holding v, link it before the current cell, and return it
        **/

        public final CLCellT addPrev(T v)
        {
                CLCellT p = prev_;
                CLCellT c = new CLCellT(v, p, this);
                p.next_ = c;
                prev_ = c;
                return c;
        }

        /**
         * link p before current cell
        **/

        public final void linkPrev(CLCellT p)
        {
                if (p !is null)
                   {
                   prev_.next_ = p;
                   p.prev_ = prev_;
                   p.next_ = this;
                   prev_ = p;
                   }
        }

        /**
         * return the number of cells in the list
        **/

        public final int _length()
        {
                int c = 0;
                CLCellT p = this;
                do {
                   ++c;
                   p = p.next();
                   } while (p !is this);
                return c;
        }

        /**
         * return the first cell holding element found in a circular traversal starting
         * at current cell, or null if no such
        **/

        public final CLCellT find(T element)
        {
                CLCellT p = this;
                do {
                   if (p.element() == (element))
                       return p;
                   p = p.next();
                   } while (p !is this);
                return null;
        }

        /**
         * return the number of cells holding element found in a circular
         * traversal
        **/

        public final int count(T element)
        {
                int c = 0;
                CLCellT p = this;
                do {
                   if (p.element() == (element))
                       ++c;
                   p = p.next();
                   } while (p !is this);
                return c;
        }

        /**
         * return the nth cell traversed from here. It may wrap around.
        **/

        public final CLCellT nth(int n)
        {
                CLCellT p = this;
                for (int i = 0; i < n; ++i)
                     p = p.next_;
                return p;
        }


        /**
         * Unlink the next cell.
         * This has no effect on the list if isSingleton()
        **/

        public final void unlinkNext()
        {
                CLCellT nn = next_.next_;
                nn.prev_ = this;
                next_ = nn;
        }

        /**
         * Unlink the previous cell.
         * This has no effect on the list if isSingleton()
        **/

        public final void unlinkPrev()
        {
                CLCellT pp = prev_.prev_;
                pp.next_ = this;
                prev_ = pp;
        }


        /**
         * Unlink self from list it is in.
         * Causes it to be a singleton
        **/

        public final void unlink()
        {
                CLCellT p = prev_;
                CLCellT n = next_;
                p.next_ = n;
                n.prev_ = p;
                prev_ = this;
                next_ = this;
        }

        /**
         * Make a copy of the list and return new head. 
        **/

        public final CLCellT copyList()
        {
                CLCellT hd = this;

                CLCellT newlist = new CLCellT(hd.element(), null, null);
                CLCellT current = newlist;

                for (CLCellT p = next_; p !is hd; p = p.next_)
                     {
                     current.next_ = new CLCellT(p.element(), current, null);
                     current = current.next_;
                     }
                newlist.prev_ = current;
                current.next_ = newlist;
                return newlist;
        }
}

alias CLCellT!(Object) CLCell;
